Examples

Define Workflows in YAML

Simple, declarative workflow definitions with powerful capabilities. 

# AI-assisted PR review with approval gate
name: guarded-pr-review
version: "1.0"

triggers:
  - type: github.pull_request
    events: [opened, synchronize]

steps:
  - name: analyze-changes
    agent: code-reviewer
    input:
      pr_number: ${{ trigger.pr.number }}
      repo: ${{ trigger.repo.full_name }}

  - name: security-scan
    agent: security-analyst
    input:
      diff: ${{ steps.analyze-changes.output.diff }}

  - name: generate-review
    agent: code-reviewer
    requires_approval: true
    input:
      analysis: ${{ steps.analyze-changes.output }}
      security: ${{ steps.security-scan.output }}

policies:
  - name: require-human-approval
    rule: steps.*.requires_approval == true
# OPA/Rego policy for workflow governance
package bridge.approval

import rego.v1

# Default: no approval required
default requires_approval := false

# Require approval if step explicitly requests it
requires_approval if {
    input.step.requires_approval == true
}

# Require approval for high-risk workflows
requires_approval if {
    input.workflow.risk_level == "high"
}

# Require approval for production deployments
requires_approval if {
    input.step.type == "deploy"
    input.environment == "production"
}

# Deny if security scan failed
deny[msg] if {
    input.security_scan.passed == false
    msg := "Security scan must pass before execution"
}
